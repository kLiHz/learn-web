JavaScript 加载 JSON

假设有下面这样的 JSON 文件：

fruits/fruits.json

{
	"fruits": [
		{
			"name": "apple",
			"color": "red"
		},
		{
			"name": "banana",
			"color": "yellow"
		},
		{
			"name": "cherries",
			"color": "red"
		}
	]
}

在 HTML 页面上有一按钮，点击后即可加载数据到 HTML 页面上的表格。

HTML 页面如下：

fruits/index.html （最终效果）

<!DOCTYPE html>
<html lang="en">

<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <link rel="stylesheet" href="stylesheet.css">
    <title>Fruit List</title>
</head>

<body>
    <h1>Fruits</h1>
    <button onclick="loadFruits()">Load Fruits</button>
    <table id="fruits_table"></table>
    <script type="text/javascript" src="load-fruits.js" defer></script>
</body>

</html>

JavaScrpt:

fruits/load-fruits.js 

var fruitsJSON = `{
	"fruits": [
		{
			"name": "apple",
			"color": "red"
		},
		{
			"name": "banana",
			"color": "yellow"
		},
		{
			"name": "cherries",
			"color": "red"
		}
	]
}`;

var fruitsData = JSON.parse(fruitsJSON);

const fruitsTableHeader = '<thead><tr><th>Name</th><th>Color</th></tr></thead>';

function loadFruits(){
    
    var fruitTableBody = "<tbody>";
    fruitsData.fruits.forEach(fruit => {
        var fruitRow = '<tr>';
        fruitRow += '<td>' + fruit.name + '</td>';
        fruitRow += '<td>' + fruit.color + '</td>';
        fruitRow += '</tr>';
        fruitTableBody += fruitRow;
    });
    fruitTableBody += '</tbody>';
    
    document.getElementById("fruits_table").innerHTML = fruitsTableHeader + fruitTableBody;
}
